%Generic Counter
\newcount\GenCounter\GenCounter=0
\def\ResetGenCounter{
\GenCounter=0 
}
\def\StepOneGenCounter{
\global\advance\GenCounter by 1
}

%Un contador de sesiones de clase
\newcount\SesionSOTRCounter\SesionSOTRCounter=0
%#1\in\{Lunes,Martes,Miercoles,\ldots\},#2La fecha, V&G 20150918
%#3 Las macros de los discentes que asistieron a cada sesion 
%de clase.

\global\advance\SesionSOTRCounter by 1
%\noindent #1\ #2\ \\

%\begin{center}
\begin{tabular}{rll}

\end{tabular}
%\end{center}
}
